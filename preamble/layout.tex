%% Adjust page layout.
\usepackage[
  paper=a4paper,
  top=3cm,
  bottom=3cm,
  left=2.5cm,
  right=2.5cm,
]{geometry}

%% Dimensions.
\usepackage{setspace}
\setstretch{1.25}
\renewcommand{\arraystretch}{1.3}
\setlength{\parskip}{0.3em plus 0.1em minus 0.1em}

%% Set up document title and author.
\title{\bfseries\doctitle}
\author{\docauthor}
\date{}

%% Set up fancy headers and footers.
\usepackage{fancyhdr}
\usepackage{lastpage}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\sffamily\nouppercase{\leftmark}}
\fancyfoot[C]{\thepage~/~\pageref{LastPage}}
\fancypagestyle{plain}{%
  \renewcommand{\headrulewidth}{0pt}%
  \fancyhead{}%
  \fancyfoot[C]{\thepage~/~\pageref{LastPage}}%
}

%% Set up title styles.
\usepackage{titlesec}

% section
\titleformat{\section}
  {\sffamily\bfseries\Large}
  {\thesection}
  {0.7em}
  {}

% subsection
\titleformat{\subsection}
  {\sffamily\bfseries\large}
  {\thesubsection}
  {0.5em}
  {}

% subsubsection
\titleformat{\subsubsection}
  {\sffamily\bfseries\centering}
  {}
  {0pt}
  {}

%% Set up toc styles.
\usepackage{xcolor}
\usepackage{titletoc}

\setcounter{tocdepth}{3}

% section
\titlecontents{section}[1.6em]
  {\addvspace{0.5em}\bfseries}
  {\contentslabel{1.4em}}{\hspace*{-1.4em}}
  {\hfill\normalfont\contentspage[\bfseries\thecontentspage]}

% subsection
\titlecontents{subsection}[4.0em]
  {\addvspace{0.2em}}
  {\contentslabel{2.3em}}{\hspace*{-2.3em}}
  {\textcolor{gray}{\titlerule*[9pt]{.}}\contentspage}

% subsubsection
\newcommand*{\sspace}{\hspace{0.34em plus 0.5em minus 0.12em}}
\titlecontents*{subsubsection}[4.7em]
  {\footnotesize\color{gray}}
  {}{}
  {~\thecontentspage}
  []
  [\sspace\textbullet\sspace][\hspace*{0.6em}\vspace{0.1em}]

% Custom macro to show the contents.
\newcommand{\makecontents}{%
  {%
    \hypersetup{hidelinks}%
    \tableofcontents%
  }%
}
